# Menerapkan Recursive Feature Elimination with Cross-Validation (RFECV) untuk Seleksi Fitur
import pandas as pd
import numpy as np
from sklearn.ensemble import RandomForestClassifier
from sklearn.feature_selection import RFECV
from sklearn.model_selection import StratifiedKFold

# 1. Muat Dataset WADI
df = pd.read_csv('WADI.csv')

# 2. Pra-pemrosesan Data: Tentukan Fitur (X) dan Target (y)
# Drop kolom waktu dan tanggal, serta kolom yang tidak relevan.
df_clean = df.drop(columns=['Date', 'Time']).copy()

# Hapus kolom non-numerik (jika ada) selain kolom target 'Attack'
# Kolom 'Attack' adalah target, sedangkan sisanya adalah fitur.
target_column = 'Attack'
X = df_clean.drop(columns=[target_column])
y = df_clean[target_column]

# Bersihkan data: RFECV tidak menangani nilai NaN, kita isi dengan 0
# Anda mungkin perlu metode imputasi yang lebih canggih di dunia nyata
X = X.fillna(0)

# 3. Inisialisasi Estimator dan RFECV
# Gunakan Random Forest Classifier sebagai estimator dasar untuk RFECV
estimator = RandomForestClassifier(n_estimators=100, random_state=42, n_jobs=-1)

# Inisialisasi RFECV: Gunakan 5-fold Stratified Cross-Validation (StratifiedKFold)
# StratifiedKFold penting karena kolom 'Attack' (target) kemungkinan tidak seimbang (imbalanced)
cv_strategy = StratifiedKFold(5) 
rfecv = RFECV(
    estimator=estimator,
    step=1,          # Jumlah fitur yang akan dieliminasi per iterasi
    cv=cv_strategy,  # Strategi cross-validation
    scoring='accuracy', # Metrik untuk mengevaluasi model
    min_features_to_select=1 # Jumlah minimum fitur yang akan dipertahankan
)

# 4. Jalankan Seleksi Fitur
print("Memulai proses RFECV...")
rfecv.fit(X, y)

# 5. Tampilkan Hasil Seleksi Fitur
optimal_feature_count = rfecv.n_features_
selected_features_mask = rfecv.support_
selected_feature_names = X.columns[selected_features_mask].tolist()

print(f"\nProses RFECV Selesai.")
print(f"Jumlah Fitur Optimal: {optimal_feature_count}")
print(f"Fitur-Fitur yang Terpilih ({optimal_feature_count} fitur):")
for feature in selected_feature_names:
    print(f"- {feature}")

# Catatan: RFECV akan mengembalikan fitur yang memberikan performa terbaik pada cross-validation.
# Plotting hasil RFECV (opsional, dapat diaktifkan di sel terpisah jika perlu visualisasi skor CV)

# Menerapkan Recursive Feature Elimination with Cross-Validation (RFECV) pada Dataset SWAT
import pandas as pd
from sklearn.ensemble import RandomForestClassifier
from sklearn.feature_selection import RFECV
from sklearn.model_selection import StratifiedKFold

# Memuat dataset SWAT.csv
df = pd.read_csv('SWAT_Dataset.csv')

# Mendefinisikan kolom target
target_column = 'Normal/Attack'
X = df.drop(columns=[target_column])
y_raw = df[target_column]

# Mengubah kolom target (string) menjadi numerik biner: Normal=0, Attack=1
y = y_raw.apply(lambda x: 1 if x.strip() == 'Attack' else 0)

# Mengatasi nilai NaN dalam fitur dengan imputasi sederhana (mengisi dengan 0)
X = X.fillna(0)

# Inisialisasi RandomForestClassifier sebagai estimator
estimator = RandomForestClassifier(n_estimators=100, random_state=42, n_jobs=-1)

# Inisialisasi RFECV dengan 5-fold Stratified Cross-Validation
cv_strategy = StratifiedKFold(5, shuffle=True, random_state=42)
rfecv = RFECV(
    estimator=estimator,
    step=1,
    cv=cv_strategy,
    scoring='accuracy', 
    min_features_to_select=1 
)

# Menjalankan proses RFECV
print("Memulai proses RFECV pada dataset SWAT...")
rfecv.fit(X, y)

# Mendapatkan hasil seleksi fitur
optimal_feature_count = rfecv.n_features_
selected_features_mask = rfecv.support_
selected_feature_names = X.columns[selected_features_mask].tolist()

# Membuat DataFrame baru dengan hanya fitur yang terpilih
X_selected = X[selected_feature_names]

# Mencetak hasil
print(f"\nProses RFECV Selesai.")
print(f"Jumlah Fitur Awal: {X.shape[1]}")
print(f"Jumlah Fitur Optimal (Hasil RFECV): {optimal_feature_count}")
print("\nDaftar Fitur yang Terpilih:")
print(selected_feature_names)
print("\nHead dari DataFrame Fitur Terpilih:")
print(X_selected.head())

# Menerapkan Recursive Feature Elimination with Cross-Validation (RFECV)
import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
from sklearn.ensemble import RandomForestClassifier
from sklearn.feature_selection import RFECV
from sklearn.model_selection import StratifiedKFold

df = pd.read_csv('WADI.csv')
target_column = 'Attack'

df_clean = df.drop(columns=['Date', 'Time']).copy()
X = df_clean.drop(columns=[target_column])
y = df_clean[target_column]

X = X.fillna(0)

estimator = RandomForestClassifier(n_estimators=100, random_state=42, n_jobs=-1)
cv_strategy = StratifiedKFold(5) 
rfecv = RFECV(
    estimator=estimator,
    step=1,
    cv=cv_strategy,
    scoring='accuracy',
    min_features_to_select=1
)

print("Memulai proses RFECV...")
rfecv.fit(X, y)

optimal_feature_count = rfecv.n_features_
selected_features_mask = rfecv.support_
selected_feature_names = X.columns[selected_features_mask].tolist()

X_selected = X[selected_feature_names]

    print(f"- {feature}")
